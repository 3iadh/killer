import { Component, OnInit, Input } from '@angular/core';
import { TranslateService } from '@ngx-translate/core';
import { Router, ActivatedRoute } from '@angular/router';
import { ContractService } from "../services/contracts.service";

import {ToastyService, ToastyConfig, ToastOptions, ToastData} from 'ng2-toasty';

@Component({
  selector: 'accueil-root',
  templateUrl: '../templates/accueil.component.html',
  styleUrls: ['../styles/accueil.component.css']
})
export class AccueilComponent {
  isConnected: boolean;
  contract = [];
  contractDetail = [];

  contratId;

  //Dans le cas ou j'ai tué quelqu'un et il faut que je déclare l'avoir tué
  killerValidationSet = false ;
  canSetKillerValidation = false ;

  //Dans le cas ou quelqu'un m'a tué et il faut que je confirme
  imKilled = false ;
  canSetImKilled = false ;

  infoDetail = {
    'targetInfo': {
      'bench': 'Espoir',
      'firstname': '',
      'killerlastename': '',
      'killerusername': ''
    },
    'contractAction': '',
    'contractPlace': '',
    'startDate': '',
    'id': null,
    'canSetKillerValidation': 0
  };

  @Input('parentData') incomingData: string;

  myUserId
  userId;
  constructor(private translate: TranslateService,
    private router: Router,
    private route: ActivatedRoute,
    private contractservice: ContractService,
    private toastyService:ToastyService, 
    private toastyConfig: ToastyConfig
  ) {
    translate.setDefaultLang('fr');
    this.toastyConfig.theme = 'material';
    // console.log(this.connectedUserId);
    // contractservice.getServerContracts();
    // contractservice.createKillerContract();
    // contractservice.getKillerContracts();

    //GetMyContract

    this.route
    .queryParams
    .subscribe(params => {
      this.userId = params['userId'];
    });
    // this.userId = this.route.snapshot.params['userId'];
    // console.log(this.userId);

    if (this.userId !== undefined) {

      this.reloadData(this.userId);

      this.getLostedContrat(this.userId);
      // contractservice.getKillerContracts().then(res => {
      //   // console.log(res.data);
      // })
    }
  }


  //permet de confirmer que nous avons tué le target
  setContratRempli() {
     let id = this.infoDetail.id ;
    if (this.infoDetail.id != null) {
      this.contractservice.setTargetKilling(id).then(res => {
        this.killerValidationSet = true ;
        this.canSetKillerValidation = false ;

        // this.addToast();

        //Recharger la page après validation
        this.reloadData(this.userId);

      })
    }
  }

  //Permet en tant que joueur de dire que j'ai été quillé

  setImKilled() {
    let id = this.infoDetail.id ;
   if (this.infoDetail.id != null) {
     this.contractservice.confirmImKilled(id).then(res => {
       this.killerValidationSet = true ;
       this.canSetKillerValidation = false ;

       this.addToast();
     })

   }
 }

//recharger les données

reloadData(userId){
  this.contractservice.getMyContract(userId).then(res => {
    this.isConnected = true;
    console.log(res.data);
    this.contract = res.data;

    //Get details of the contract
    this.contractservice.getContract(this.contract[0].contractid.data.id).then(rep => {
      this.contractDetail = rep.data;
      // console.log(this.contractDetail[0]);

    //  this.canSetKillerValidation = true ;
      this.infoDetail = {
        'targetInfo': {
          'bench': this.contractDetail[0].contacttargetid.data.killerbench,
          'firstname': this.contractDetail[0].contacttargetid.data.killerfirstname,
          'killerlastename': this.contractDetail[0].contacttargetid.data.killerlastename,
          'killerusername': this.contractDetail[0].contacttargetid.data.killersurname,
        },
        'contractAction': this.contractDetail[0].contractactionid.data.actiondescription,
        'contractPlace': this.contractDetail[0].contractplaceid.data.placedescription,
        'startDate': this.contractDetail[0].contractstartdate,
        'id': this.contract[0].id,
        'canSetKillerValidation' : this.contract[0].killervalidation
      }

      // console.log(this.infoDetail);

    }).catch(err => {

    })

  }).catch(res => {

  });
}
